107 def on(self, event, handler=None, namespace=None):
108 """Register an event handler.
109
110 :param event: The event name. It can be any string. The event names
111 ``'connect'``, ``'message'`` and ``'disconnect'`` are
112 reserved and should not be used. The ``'*'`` event name
113 can be used to define a catch-all event handler.
114 :param handler: The function that should be invoked to handle the
115 event. When this parameter is not given, the method
116 acts as a decorator for the handler function.
117 :param namespace: The Socket.IO namespace for the event. If this
118 argument is omitted the handler is associated with
119 the default namespace. A catch-all namespace can be
120 defined by passing ``'*'`` as the namespace.
121
122 Example usage::
123
124 # as a decorator:
125 @sio.on('connect')
126 def connect_handler():
127 print('Connected!')
128
129 # as a method:
130 def message_handler(msg):
131 print('Received message: ', msg)
132 sio.send( 'response')
133 sio.on('message', message_handler)
134
135 The arguments passed to the handler function depend on the event type:
136
137 - The ``'connect'`` event handler does not take arguments.
138 - The ``'disconnect'`` event handler does not take arguments.
139 - The ``'message'`` handler and handlers for custom event names receive
140 the message payload as only argument. Any values returned from a
141 message handler will be passed to the client's acknowledgement
142 callback function if it exists.
143 - A catch-all event handler receives the event name as first argument,
144 followed by any arguments specific to the event.
145 - A catch-all namespace event handler receives the namespace as first
146 argument, followed by any arguments specific to the event.
147 - A combined catch-all namespace and catch-all event handler receives
148 the event name as first argument and the namespace as second
149 argument, followed by any arguments specific to the event.
150 """
151 namespace = namespace or '/'
152
153 def set_handler(handler):
154 if namespace not in self.handlers:
155 self.handlers[namespace] = {}
156 self.handlers[namespace][event] = handler
157 return handler
158
159 if handler is None:
160 return set_handler
161 set_handler(handler)

163 def event(self, *args, **kwargs):
164 """Decorator to register an event handler.
165
166 This is a simplified version of the ``on()`` method that takes the
167 event name from the decorated function.
168
169 Example usage::
170
171 @sio.event
172 def my_event(data):
173 print('Received data: ', data)
174
175 The above example is equivalent to::
176
177 @sio.on('my_event')
178 def my_event(data):
179 print('Received data: ', data)
180
181 A custom namespace can be given as an argument to the decorator::
182
183 @sio.event(namespace='/test')
184 def my_event(data):
185 print('Received data: ', data)
186 """
187 if len(args) == 1 and len(kwargs) == 0 and callable(args[0]):
188 # the decorator was invoked without arguments
189 # args[0] is the decorated function
190 return self.on(args[0].__name__)(args[0])
191 else:
192 # the decorator was invoked with arguments
193 def set_handler(handler):
194 return self.on(handler.__name__, *args, **kwargs)(handler)
195
196 return set_handler

236 def _get_event_handler(self, event, namespace, args):
237 # return the appropriate application event handler
238 #
239 # Resolution priority:
240 # - self.handlers[namespace][event]
241 # - self.handlers[namespace]["*"]
242 # - self.handlers["*"][event]
243 # - self.handlers["*"]["*"]
244 handler = None
245 if namespace in self.handlers:
246 if event in self.handlers[namespace]:
247 handler = self.handlers[namespace][event]
248 elif event not in self.reserved_events and \
249 '*' in self.handlers[namespace]:
250 handler = self.handlers[namespace]['*']
251 args = (event, *args)
252 elif '*' in self.handlers:
253 if event in self.handlers['*']:
254 handler = self.handlers['*'][event]
255 args = (namespace, *args)
256 elif event not in self.reserved_events and \
257 '*' in self.handlers['*']:
258 handler = self.handlers['*']['*']
259 args = (event, namespace, *args)
260 return handler, args
