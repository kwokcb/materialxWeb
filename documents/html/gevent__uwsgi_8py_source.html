38 if hasattr(uwsgi, 'request_context'):
39 # uWSGI >= 2.1.x with support for api access across-greenlets

46 # spawn a select greenlet
47 def select_greenlet_runner(fd, event):
48 """Sets event when data becomes available to read on fd."""
49 sel = selectors.DefaultSelector()
50 sel.register(fd, selectors.EVENT_READ)
51 try:
52 while True:
53 sel.select()
54 event.set()
55 except gevent.GreenletExit:
56 sel.unregister(fd)
57 self._select_greenlet = gevent.spawn(
58 select_greenlet_runner,
59 self._sock,
60 self._event)

64 return '' # send nothing as response

66 def close(self):
67 """Disconnects uWSGI from the client."""
68 if self._req_ctx is None:
69 # better kill it here in case wait() is not called again
70 self._select_greenlet.kill()
71 self._event.set()

73 def _send(self, msg):
74 """Transmits message either in binary or UTF-8 text mode,
75 depending on its type."""
76 if isinstance(msg, bytes):
77 method = uwsgi.websocket_send_binary
78 else:
79 method = uwsgi.websocket_send
80 if self._req_ctx is not None:
81 method(msg, request_context=self._req_ctx)
82 else:
83 method(msg)
84
85 def _decode_received(self, msg):
86 """Returns either bytes or str, depending on message type."""
87 if not isinstance(msg, bytes):
88 # already decoded - do nothing
89 return msg
90 # only decode from utf-8 if message is not binary data
91 type = ord(msg[0:1])
92 if type >= 48: # no binary
93 return msg.decode('utf-8')
94 # binary message, don't try to decode
95 return msg

97 def send(self, msg):
98 """Queues a message for sending. Real transmission is done in
99 wait method.
100 Sends directly if uWSGI version is new enough."""
101 if self._req_ctx is not None:
102 self._send(msg)
103 else:
104 self._send_queue.put(msg)
105 self._event.set()

107 def wait(self):
108 """Waits and returns received messages.
109 If running in compatibility mode for older uWSGI versions,
110 it also sends messages that have been queued by send().
111 A return value of None means that connection was closed.
112 This must be called repeatedly. For uWSGI < 2.1.x it must
113 be called from the main greenlet."""
114 while True:
115 if self._req_ctx is not None:
116 try:
117 msg = uwsgi.websocket_recv(request_context=self._req_ctx)
118 except OSError: # connection closed
119 self.close()
120 return None
121 return self._decode_received(msg)
122 else:
123 if self.received_messages:
124 return self.received_messages.pop(0)
125
126 # we wake up at least every 3 seconds to let uWSGI
127 # do its ping/ponging
128 event_set = self._event.wait(timeout=3)
129 if event_set:
130 self._event.clear()
131 # maybe there is something to send
132 msgs = []
133 while True:
134 try:
135 msgs.append(self._send_queue.get(block=False))
136 except gevent.queue.Empty:
137 break
138 for msg in msgs:
139 try:
140 self._send(msg)
141 except OSError:
142 self.close()
143 return None
144 # maybe there is something to receive, if not, at least
145 # ensure uWSGI does its ping/ponging
146 while True:
147 try:
148 msg = uwsgi.websocket_recv_nb()
149 except OSError: # connection closed
150 self.close()
151 return None
152 if msg: # message available
153 self.received_messages.append(
154 self._decode_received(msg))
155 else:
156 break
157 if self.received_messages:
158 return self.received_messages.pop(0)
