169class Context:
170 """The context is a special internal object that holds state relevant
171 for the script execution at every single level. It's normally invisible
172 to commands unless they opt-in to getting access to it.
173
174 The context is useful as it can pass internal objects around and can
175 control special execution features such as reading data from
176 environment variables.
177
178 A context can be used as context manager in which case it will call
179 :meth:`close` on teardown.
180
181 :param command: the command class for this context.
182 :param parent: the parent context.
183 :param info_name: the info name for this invocation. Generally this
184 is the most descriptive name for the script or
185 command. For the toplevel script it is usually
186 the name of the script, for commands below it it's
187 the name of the script.
188 :param obj: an arbitrary object of user data.
189 :param auto_envvar_prefix: the prefix to use for automatic environment
190 variables. If this is `None` then reading
191 from environment variables is disabled. This
192 does not affect manually set environment
193 variables which are always read.
194 :param default_map: a dictionary (like object) with default values
195 for parameters.
196 :param terminal_width: the width of the terminal. The default is
197 inherit from parent context. If no context
198 defines the terminal width then auto
199 detection will be applied.
200 :param max_content_width: the maximum width for content rendered by
201 Click (this currently only affects help
202 pages). This defaults to 80 characters if
203 not overridden. In other words: even if the
204 terminal is larger than that, Click will not
205 format things wider than 80 characters by
206 default. In addition to that, formatters might
207 add some safety mapping on the right.
208 :param resilient_parsing: if this flag is enabled then Click will
209 parse without any interactivity or callback
210 invocation. Default values will also be
211 ignored. This is useful for implementing
212 things such as completion support.
213 :param allow_extra_args: if this is set to `True` then extra arguments
214 at the end will not raise an error and will be
215 kept on the context. The default is to inherit
216 from the command.
217 :param allow_interspersed_args: if this is set to `False` then options
218 and arguments cannot be mixed. The
219 default is to inherit from the command.
220 :param ignore_unknown_options: instructs click to ignore options it does
221 not know and keeps them for later
222 processing.
223 :param help_option_names: optionally a list of strings that define how
224 the default help parameter is named. The
225 default is ``['--help']``.
226 :param token_normalize_func: an optional function that is used to
227 normalize tokens (options, choices,
228 etc.). This for instance can be used to
229 implement case insensitive behavior.
230 :param color: controls if the terminal supports ANSI colors or not. The
231 default is autodetection. This is only needed if ANSI
232 codes are used in texts that Click prints which is by
233 default not the case. This for instance would affect
234 help output.
235 :param show_default: Show the default value for commands. If this
236 value is not set, it defaults to the value from the parent
237 context. ``Command.show_default`` overrides this default for the
238 specific command.
239
240 .. versionchanged:: 8.1
241 The ``show_default`` parameter is overridden by
242 ``Command.show_default``, instead of the other way around.
243
244 .. versionchanged:: 8.0
245 The ``show_default`` parameter defaults to the value from the
246 parent context.
247
248 .. versionchanged:: 7.1
249 Added the ``show_default`` parameter.
250
251 .. versionchanged:: 4.0
252 Added the ``color``, ``ignore_unknown_options``, and
253 ``max_content_width`` parameters.
254
255 .. versionchanged:: 3.0
256 Added the ``allow_extra_args`` and ``allow_interspersed_args``
257 parameters.
258
259 .. versionchanged:: 2.0
260 Added the ``resilient_parsing``, ``help_option_names``, and
261 ``token_normalize_func`` parameters.
262 """

478 @contextmanager
479 def scope(self, cleanup: bool = True) -> t.Iterator["Context"]:
480 """This helper method can be used with the context object to promote
481 it to the current thread local (see :func:`get_current_context`).
482 The default behavior of this is to invoke the cleanup functions which
483 can be disabled by setting `cleanup` to `False`. The cleanup
484 functions are typically used for things such as closing file handles.
485
486 If the cleanup is intended the context object can also be directly
487 used as a context manager.
488
489 Example usage::
490
491 with ctx.scope():
492 assert get_current_context() is ctx
493
494 This is equivalent::
495
496 with ctx:
497 assert get_current_context() is ctx
498
499 .. versionadded:: 5.0
500
501 :param cleanup: controls if the cleanup functions should be run or
502 not. The default is to run these functions. In
503 some situations the context only wants to be
504 temporarily pushed in which case this can be disabled.
505 Nested pushes automatically defer the cleanup.
506 """
507 if not cleanup:
508 self._depth += 1
509 try:
510 with self as rv:
511 yield rv
512 finally:
513 if not cleanup:
514 self._depth -= 1

583 .. versionadded:: 8.0

743 """Invokes a command callback in exactly the way it expects. There
744 are two ways to invoke this method:
745
746 1. the first argument can be a callback and all other arguments and
747 keyword arguments are forwarded directly to the function.
748 2. the first argument is a click command object. In that case all
749 arguments are forwarded as well but proper click parameters
750 (options and click arguments) must be keyword arguments and Click
751 will fill in defaults.
752
753 Note that before Click 3.2 keyword arguments were not properly filled
754 in against the intention of this code and no context was created. For
755 more information about this change and why it was done in a bugfix
756 release see :ref:`upgrade-to-3.2`.
757
758 .. versionchanged:: 8.0
759 All ``kwargs`` are tracked in :attr:`params` so they will be
760 passed if :meth:`forward` is called at multiple levels.
761 """
762 if isinstance(__callback, Command):
763 other_cmd = __callback
764
765 if other_cmd.callback is None:
766 raise TypeError(
767 "The given command does not have a callback that can be invoked."
768 )
769 else:
770 __callback = t.cast("t.Callable[..., V]", other_cmd.callback)
771
772 ctx = __self._make_sub_context(other_cmd)
773
774 for param in other_cmd.params:
775 if param.name not in kwargs and param.expose_value:
776 kwargs[param.name] = param.type_cast_value( # type: ignore
777 ctx, param.get_default(ctx)
778 )
779
780 # Track all kwargs as params, so that forward() will pass
781 # them on in subsequent calls.
782 ctx.params.update(kwargs)
783 else:
784 ctx = __self
785
786 with augment_usage_errors(__self):
787 with ctx:
788 return __callback(*args, **kwargs)

790 def forward(__self, __cmd: "Command", *args: t.Any, **kwargs: t.Any) -> t.Any:
791 """Similar to :meth:`invoke` but fills in default keyword
792 arguments from the current context if the other command expects
793 it. This cannot invoke callbacks directly, only other commands.
794
795 .. versionchanged:: 8.0
796 All ``kwargs`` are tracked in :attr:`params` so they will be
797 passed if ``forward`` is called at multiple levels.
798 """
799 # Can only forward to other commands, not direct callbacks.
800 if not isinstance(__cmd, Command):
801 raise TypeError("Callback is not a command.")
802
803 for param in __self.params:
804 if param not in kwargs:
805 kwargs[param] = __self.params[param]
806
807 return __self.invoke(__cmd, *args, **kwargs)

1023 """This is the way to invoke a script with all the bells and
1024 whistles as a command line application. This will always terminate
1025 the application after a call. If this is not wanted, ``SystemExit``
1026 needs to be caught.
1027
1028 This method is also available by directly calling the instance of
1029 a :class:`Command`.
1030
1031 :param args: the arguments that should be used for parsing. If not
1032 provided, ``sys.argv[1:]`` is used.
1033 :param prog_name: the program name that should be used. By default
1034 the program name is constructed by taking the file
1035 name from ``sys.argv[0]``.
1036 :param complete_var: the environment variable that controls the
1037 bash completion support. The default is
1038 ``"_<prog_name>_COMPLETE"`` with prog_name in
1039 uppercase.
1040 :param standalone_mode: the default behavior is to invoke the script
1041 in standalone mode. Click will then
1042 handle exceptions and convert them into
1043 error messages and the function will never
1044 return but shut down the interpreter. If
1045 this is set to `False` they will be
1046 propagated to the caller and the return
1047 value of this function is the return value
1048 of :meth:`invoke`.
1049 :param windows_expand_args: Expand glob patterns, user dir, and
1050 env vars in command line args on Windows.
1051 :param extra: extra keyword arguments are forwarded to the context
1052 constructor. See :class:`Context` for more information.

1056 disabling command line arg expansion on Windows.
1057

1059 When taking arguments from ``sys.argv`` on Windows, glob
1060 patterns, user dir, and env vars are expanded.
1061
1062 .. versionchanged:: 3.0
1063 Added the ``standalone_mode`` parameter.
1064 """
1065 if args is None:
1066 args = sys.argv[1:]
1067
1068 if os.name == "nt" and windows_expand_args:
1069 args = _expand_args(args)
1070 else:
1071 args = list(args)
1072
1073 if prog_name is None:
1074 prog_name = _detect_program_name()
1075
1076 # Process shell completion requests and exit early.
1077 self._main_shell_completion(extra, prog_name, complete_var)
1078
1079 try:
1080 try:
1081 with self.make_context(prog_name, args, **extra) as ctx:
1082 rv = self.invoke(ctx)
1083 if not standalone_mode:
1084 return rv
1085 # it's not safe to `ctx.exit(rv)` here!
1086 # note that `rv` may actually contain data like "1" which
1087 # has obvious effects
1088 # more subtle case: `rv=[None, None]` can come out of
1089 # chained commands which all returned `None` -- so it's not
1090 # even always obvious that `rv` indicates success/failure
1091 # by its truthiness/falsiness
1092 ctx.exit()
1093 except (EOFError, KeyboardInterrupt) as e:
1094 echo(file=sys.stderr)
1095 raise Abort() from e
1096 except ClickException as e:
1097 if not standalone_mode:
1098 raise
1099 e.show()
1100 sys.exit(e.exit_code)
1101 except OSError as e:
1102 if e.errno == errno.EPIPE:
1103 sys.stdout = t.cast(t.TextIO, PacifyFlushWrapper(sys.stdout))
1104 sys.stderr = t.cast(t.TextIO, PacifyFlushWrapper(sys.stderr))
1105 sys.exit(1)
1106 else:
1107 raise
1108 except Exit as e:
1109 if standalone_mode:
1110 sys.exit(e.exit_code)
1111 else:
1112 # in non-standalone mode, return the exit code
1113 # note that this is only reached if `self.invoke` above raises
1114 # an Exit explicitly -- thus bypassing the check there which
1115 # would return its result
1116 # the results of non-standalone execution may therefore be
1117 # somewhat ambiguous: if there are codepaths which lead to
1118 # `ctx.exit(1)` and to `return 1`, the caller won't be able to
1119 # tell the difference between the two
1120 return e.exit_code
1121 except Abort:
1122 if not standalone_mode:
1123 raise
1124 echo(_("Aborted!"), file=sys.stderr)
1125 sys.exit(1)
