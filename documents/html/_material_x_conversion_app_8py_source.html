105 def handle_load_materialx(self, data):
106 '''
107 Handle loading in of MaterialX document
108 '''
109 materialx_file = data.get('materialxFile', 'Default MaterialX File')
110 print('> Server: load materialx event received. File:', materialx_file)
111 materialx_content = data.get('content', 'MaterialX content')
112 doc = mx.createDocument()
113 if len(materialx_content) == 0:
114 return
115 mx.readFromXmlString(doc, materialx_content)
116 print('>> MaterialX Document loaded')
117 doc_string = mx.writeToXmlString(doc)
118 emit('materialx_loaded', {'materialxDocument': doc_string}, broadcast=True)

120 def handle_render_materialx(self, data):
121 '''
122 Handle request to render MaterialX document
123 '''
124 materialx_string = data.get('materialxDocument', 'MaterialX content')
125 print('> Server: render_materialx event received')
126
127 # Get environment variable: MATERIALX_DEFAULT_VIEWER
128 ilm_viewer = os.getenv('MATERIALX_DEFAULT_VIEWER', '')
129 if len(ilm_viewer) == 0:
130 print('>> MATERIALX_DEFAULT_VIEWER environment variable not set')
131 return
132
133 doc = mx.createDocument()
134 if len(materialx_string) == 0:
135 return
136
137 stdlib = mx.createDocument()
138 mx.loadLibraries(mx.getDefaultDataLibraryFolders(), mx.getDefaultDataSearchPath(), stdlib)
139 doc.importLibrary(stdlib)
140 mx.readFromXmlString(doc, materialx_string)
141
142 # Get platform temporary folder location
143 temp_location = os.getenv('TEMP', '/tmp')
144 if not os.path.exists(temp_location):
145 temp_location = '/tmp'
146 # Generate a temp file name based on the current date and time
147 temp_name = datetime.datetime.now().strftime('%Y%m%d%H%M%S') + '.mtlx'
148 temp_file = f'{temp_location}/{temp_name}'
149 mx.writeToXmlFile(doc, temp_file)
150 capture_filename = temp_file.replace(".mtlx", ".png")
151 cmd = f'{ilm_viewer} --screenWidth 512 --screenHeight 512 '
152 cmd += f' --captureFilename {capture_filename} --material {temp_file}'
153 print('>> Rendering:', cmd)
154 os.system(cmd)
155 # Delete the temp files
156 os.remove(temp_file)
157
158 image_base64 = self.convert_png_to_base64(capture_filename)
159 os.remove(capture_filename)
160 print('>> Emit materialx_rendered event')
161 emit('materialx_rendered', {'image': image_base64}, broadcast=True)

163 def handle_convert_to_usd(self, data):
164 '''
165 Handle request to convert MaterialX to USD
166 '''
167 if not have_usd_converter:
168 emit('usd_converted', {'usdDocument': ''}, broadcast=True)
169 return
170
171 materialx_string = data.get('materialxDocument', '')
172 print('> Server: convert-to-usd event received')
173 doc = mx.createDocument()
174 if len(materialx_string) == 0:
175 return
176 try:
177 mx.readFromXmlString(doc, materialx_string)
178 print('>> MaterialX document loaded')
179 stage_string = convertMtlxToUsd(doc, True)
180 print('>> USD stage created.')
181 emit('usd_converted', {'usdDocument': stage_string}, broadcast=True)
182 except Exception as e:
183 print(f'>> Error during USD conversion: {e}')
184 emit('usd_converted', {'usdDocument': ''}, broadcast=True)

186 def handle_convert_to_glTF(self, data):
187 '''
188 Handle request to convert MaterialX to glTF Texture Procedural
189 graph
190 '''
191 if not have_gltf_converter:
192 emit('gltf_converted', {'document': '{}'}, broadcast=True)
193 return
194
195 materialx_string = data.get('materialxDocument', '')
196 print('> Server: convert_to_glTF event received')
197 stdlib, _ = MxGLTFPTUtil.load_standard_libraries()
198 doc = MxGLTFPTUtil.create_working_document([stdlib])
199 if len(materialx_string) == 0:
200 return
201 mx.readFromXmlString(doc, materialx_string)
202 print('>> MaterialX document loaded')
203 converter = MxGLTFPT.glTFMaterialXConverter()
204 json_string, status = converter.materialX_to_glTF(doc)
205 if not json_string:
206 print('>> Error converting to glTF:', status)
207 json_string = '{}'
208 print('>> glTF JSON created:')
209 emit('gltf_converted', {'document': json_string}, broadcast=True)
