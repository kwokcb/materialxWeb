120 def on(self, event, handler=None):
121 """Register an event handler.
122
123 :param event: The event name. Can be ``'connect'``, ``'message'`` or
124 ``'disconnect'``.
125 :param handler: The function that should be invoked to handle the
126 event. When this parameter is not given, the method
127 acts as a decorator for the handler function.
128
129 Example usage::
130
131 # as a decorator:
132 @eio.on('connect')
133 def connect_handler(sid, environ):
134 print('Connection request')
135 if environ['REMOTE_ADDR'] in blacklisted:
136 return False # reject
137
138 # as a method:
139 def message_handler(sid, msg):
140 print('Received message: ', msg)
141 eio.send(sid, 'response')
142 eio.on('message', message_handler)
143
144 The handler function receives the ``sid`` (session ID) for the
145 client as first argument. The ``'connect'`` event handler receives the
146 WSGI environment as a second argument, and can return ``False`` to
147 reject the connection. The ``'message'`` handler receives the message
148 payload as a second argument. The ``'disconnect'`` handler does not
149 take a second argument.
150 """
151 if event not in self.event_names:
152 raise ValueError('Invalid event')
