21def translate_request(request):
22 """This function takes the arguments passed to the request handler and
23 uses them to generate a WSGI compatible environ dictionary.
24 """
25 message = request._message
26 payload = request._payload
27
28 uri_parts = urlsplit(message.path)
29 environ = {
30 'wsgi.input': payload,
31 'wsgi.errors': sys.stderr,
32 'wsgi.version': (1, 0),
33 'wsgi.async': True,
34 'wsgi.multithread': False,
35 'wsgi.multiprocess': False,
36 'wsgi.run_once': False,
37 'SERVER_SOFTWARE': 'aiohttp',
38 'REQUEST_METHOD': message.method,
39 'QUERY_STRING': uri_parts.query or '',
40 'RAW_URI': message.path,
41 'SERVER_PROTOCOL': 'HTTP/%s.%s' % message.version,
42 'REMOTE_ADDR': '127.0.0.1',
43 'REMOTE_PORT': '0',
44 'SERVER_NAME': 'aiohttp',
45 'SERVER_PORT': '0',
46 'aiohttp.request': request
47 }
48
49 for hdr_name, hdr_value in message.headers.items():
50 hdr_name = hdr_name.upper()
51 if hdr_name == 'CONTENT-TYPE':
52 environ['CONTENT_TYPE'] = hdr_value
53 continue
54 elif hdr_name == 'CONTENT-LENGTH':
55 environ['CONTENT_LENGTH'] = hdr_value
56 continue
57
58 key = 'HTTP_%s' % hdr_name.replace('-', '_')
59 if key in environ:
60 hdr_value = f'{environ[key]},{hdr_value}'
61
62 environ[key] = hdr_value
63
64 environ['wsgi.url_scheme'] = environ.get('HTTP_X_FORWARDED_PROTO', 'http')
65
66 path_info = uri_parts.path
67
68 environ['PATH_INFO'] = path_info
69 environ['SCRIPT_NAME'] = ''
70
71 return environ
