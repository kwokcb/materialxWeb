132 def _safe_data_files(self, build_py):
133 """
134 Since the ``sdist`` class is also used to compute the MANIFEST
135 (via :obj:`setuptools.command.egg_info.manifest_maker`),
136 there might be recursion problems when trying to obtain the list of
137 data_files and ``include_package_data=True`` (which in turn depends on
138 the files included in the MANIFEST).
139
140 To avoid that, ``manifest_maker`` should be able to overwrite this
141 method and avoid recursive attempts to build/analyze the MANIFEST.
142 """
143 return build_py.data_files

198 def read_manifest(self):
199 """Read the manifest file (named by 'self.manifest') and use it to
200 fill in 'self.filelist', the list of files to include in the source
201 distribution.
202 """
203 log.info("reading manifest file '%s'", self.manifest)
204 manifest = open(self.manifest, 'rb')
205 for bytes_line in manifest:
206 # The manifest must contain UTF-8. See #303.
207 try:
208 line = bytes_line.decode('UTF-8')
209 except UnicodeDecodeError:
210 log.warn("%r not UTF-8 decodable -- skipping" % line)
211 continue
212 # ignore comments and blank lines
213 line = line.strip()
214 if line.startswith('#') or not line:
215 continue
216 self.filelist.append(line)
217 manifest.close()
