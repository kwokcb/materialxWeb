16class OnDupAction(Enum):
17 """An action to take to prevent duplication from occurring."""

35class OnDup(t.NamedTuple):
36 r"""A combination of :class:`~bidict.OnDupAction`\s specifying how to handle various types of duplication.
37
38 The :attr:`~OnDup.key` field specifies what action to take when a duplicate key is encountered.
39
40 The :attr:`~OnDup.val` field specifies what action to take when a duplicate value is encountered.
41
42 In the case of both key and value duplication across two different items,
43 only :attr:`~OnDup.val` is used.
44
45 *See also* :ref:`basic-usage:Values Must Be Unique`
46 (https://bidict.rtfd.io/basic-usage.html#values-must-be-unique)
47 """
48
49 key: OnDupAction = DROP_OLD
50 val: OnDupAction = RAISE
51
52
53#: Default :class:`OnDup` used for the
54#: :meth:`~bidict.bidict.__init__`,
55#: :meth:`~bidict.bidict.__setitem__`, and
56#: :meth:`~bidict.bidict.update` methods.
57ON_DUP_DEFAULT: t.Final[OnDup] = OnDup(key=DROP_OLD, val=RAISE)
58#: An :class:`OnDup` whose members are all :obj:`RAISE`.
59ON_DUP_RAISE: t.Final[OnDup] = OnDup(key=RAISE, val=RAISE)
60#: An :class:`OnDup` whose members are all :obj:`DROP_OLD`.
61ON_DUP_DROP_OLD: t.Final[OnDup] = OnDup(key=DROP_OLD, val=DROP_OLD)
