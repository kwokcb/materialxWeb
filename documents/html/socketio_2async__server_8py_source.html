16class AsyncServer(base_server.BaseServer):
17 """A Socket.IO server for asyncio.
18
19 This class implements a fully compliant Socket.IO web server with support
20 for websocket and long-polling transports, compatible with the asyncio
21 framework.
22
23 :param client_manager: The client manager instance that will manage the
24 client list. When this is omitted, the client list
25 is stored in an in-memory structure, so the use of
26 multiple connected servers is not possible.
27 :param logger: To enable logging set to ``True`` or pass a logger object to
28 use. To disable logging set to ``False``. Note that fatal
29 errors are logged even when ``logger`` is ``False``.
30 :param json: An alternative json module to use for encoding and decoding
31 packets. Custom json modules must have ``dumps`` and ``loads``
32 functions that are compatible with the standard library
33 versions.
34 :param async_handlers: If set to ``True``, event handlers for a client are
35 executed in separate threads. To run handlers for a
36 client synchronously, set to ``False``. The default
37 is ``True``.
38 :param always_connect: When set to ``False``, new connections are
39 provisory until the connect handler returns
40 something other than ``False``, at which point they
41 are accepted. When set to ``True``, connections are
42 immediately accepted, and then if the connect
43 handler returns ``False`` a disconnect is issued.
44 Set to ``True`` if you need to emit events from the
45 connect handler and your client is confused when it
46 receives events before the connection acceptance.
47 In any other case use the default of ``False``.
48 :param namespaces: a list of namespaces that are accepted, in addition to
49 any namespaces for which handlers have been defined. The
50 default is `['/']`, which always accepts connections to
51 the default namespace. Set to `'*'` to accept all
52 namespaces.
53 :param kwargs: Connection parameters for the underlying Engine.IO server.
54
55 The Engine.IO configuration supports the following settings:
56
57 :param async_mode: The asynchronous model to use. See the Deployment
58 section in the documentation for a description of the
59 available options. Valid async modes are "aiohttp",
60 "sanic", "tornado" and "asgi". If this argument is not
61 given, "aiohttp" is tried first, followed by "sanic",
62 "tornado", and finally "asgi". The first async mode that
63 has all its dependencies installed is the one that is
64 chosen.
65 :param ping_interval: The interval in seconds at which the server pings
66 the client. The default is 25 seconds. For advanced
67 control, a two element tuple can be given, where
68 the first number is the ping interval and the second
69 is a grace period added by the server.
70 :param ping_timeout: The time in seconds that the client waits for the
71 server to respond before disconnecting. The default
72 is 20 seconds.
73 :param max_http_buffer_size: The maximum size that is accepted for incoming
74 messages. The default is 1,000,000 bytes. In
75 spite of its name, the value set in this
76 argument is enforced for HTTP long-polling and
77 WebSocket connections.
78 :param allow_upgrades: Whether to allow transport upgrades or not. The
79 default is ``True``.
80 :param http_compression: Whether to compress packages when using the
81 polling transport. The default is ``True``.
82 :param compression_threshold: Only compress messages when their byte size
83 is greater than this value. The default is
84 1024 bytes.
85 :param cookie: If set to a string, it is the name of the HTTP cookie the
86 server sends back to the client containing the client
87 session id. If set to a dictionary, the ``'name'`` key
88 contains the cookie name and other keys define cookie
89 attributes, where the value of each attribute can be a
90 string, a callable with no arguments, or a boolean. If set
91 to ``None`` (the default), a cookie is not sent to the
92 client.
93 :param cors_allowed_origins: Origin or list of origins that are allowed to
94 connect to this server. Only the same origin
95 is allowed by default. Set this argument to
96 ``'*'`` to allow all origins, or to ``[]`` to
97 disable CORS handling.
98 :param cors_credentials: Whether credentials (cookies, authentication) are
99 allowed in requests to this server. The default is
100 ``True``.
101 :param monitor_clients: If set to ``True``, a background task will ensure
102 inactive clients are closed. Set to ``False`` to
103 disable the monitoring task (not recommended). The
104 default is ``True``.
105 :param transports: The list of allowed transports. Valid transports
106 are ``'polling'`` and ``'websocket'``. Defaults to
107 ``['polling', 'websocket']``.
108 :param engineio_logger: To enable Engine.IO logging set to ``True`` or pass
109 a logger object to use. To disable logging set to
110 ``False``. The default is ``False``. Note that
111 fatal errors are logged even when
112 ``engineio_logger`` is ``False``.
113 """

129 async def emit(self, event, data=None, to=None, room=None, skip_sid=None,
130 namespace=None, callback=None, ignore_queue=False):
131 """Emit a custom event to one or more connected clients.
132
133 :param event: The event name. It can be any string. The event names
134 ``'connect'``, ``'message'`` and ``'disconnect'`` are
135 reserved and should not be used.
136 :param data: The data to send to the client or clients. Data can be of
137 type ``str``, ``bytes``, ``list`` or ``dict``. To send
138 multiple arguments, use a tuple where each element is of
139 one of the types indicated above.
140 :param to: The recipient of the message. This can be set to the
141 session ID of a client to address only that client, to any
142 any custom room created by the application to address all
143 the clients in that room, or to a list of custom room
144 names. If this argument is omitted the event is broadcasted
145 to all connected clients.
146 :param room: Alias for the ``to`` parameter.
147 :param skip_sid: The session ID of a client to skip when broadcasting
148 to a room or to all clients. This can be used to
149 prevent a message from being sent to the sender.
150 :param namespace: The Socket.IO namespace for the event. If this
151 argument is omitted the event is emitted to the
152 default namespace.
153 :param callback: If given, this function will be called to acknowledge
154 the client has received the message. The arguments
155 that will be passed to the function are those provided
156 by the client. Callback functions can only be used
157 when addressing an individual client.
158 :param ignore_queue: Only used when a message queue is configured. If
159 set to ``True``, the event is emitted to the
160 clients directly, without going through the queue.
161 This is more efficient, but only works when a
162 single server process is used. It is recommended
163 to always leave this parameter with its default
164 value of ``False``.
165
166 Note: this method is not designed to be used concurrently. If multiple
167 tasks are emitting at the same time to the same client connection, then
168 messages composed of multiple packets may end up being sent in an
169 incorrect sequence. Use standard concurrency solutions (such as a Lock
170 object) to prevent this situation.
171
172 Note 2: this method is a coroutine.
173 """
174 namespace = namespace or '/'
175 room = to or room
176 self.logger.info('emitting event "%s" to %s [%s]', event,
177 room or 'all', namespace)
178 await self.manager.emit(event, data, namespace, room=room,
179 skip_sid=skip_sid, callback=callback,
180 ignore_queue=ignore_queue)

225 async def call(self, event, data=None, to=None, sid=None, namespace=None,
226 timeout=60, ignore_queue=False):
227 """Emit a custom event to a client and wait for the response.
228
229 This method issues an emit with a callback and waits for the callback
230 to be invoked before returning. If the callback isn't invoked before
231 the timeout, then a ``TimeoutError`` exception is raised. If the
232 Socket.IO connection drops during the wait, this method still waits
233 until the specified timeout.
234
235 :param event: The event name. It can be any string. The event names
236 ``'connect'``, ``'message'`` and ``'disconnect'`` are
237 reserved and should not be used.
238 :param data: The data to send to the client or clients. Data can be of
239 type ``str``, ``bytes``, ``list`` or ``dict``. To send
240 multiple arguments, use a tuple where each element is of
241 one of the types indicated above.
242 :param to: The session ID of the recipient client.
243 :param sid: Alias for the ``to`` parameter.
244 :param namespace: The Socket.IO namespace for the event. If this
245 argument is omitted the event is emitted to the
246 default namespace.
247 :param timeout: The waiting timeout. If the timeout is reached before
248 the client acknowledges the event, then a
249 ``TimeoutError`` exception is raised.
250 :param ignore_queue: Only used when a message queue is configured. If
251 set to ``True``, the event is emitted to the
252 client directly, without going through the queue.
253 This is more efficient, but only works when a
254 single server process is used. It is recommended
255 to always leave this parameter with its default
256 value of ``False``.
257
258 Note: this method is not designed to be used concurrently. If multiple
259 tasks are emitting at the same time to the same client connection, then
260 messages composed of multiple packets may end up being sent in an
261 incorrect sequence. Use standard concurrency solutions (such as a Lock
262 object) to prevent this situation.
263
264 Note 2: this method is a coroutine.
265 """
266 if to is None and sid is None:
267 raise ValueError('Cannot use call() to broadcast.')
268 if not self.async_handlersasync_handlers:
269 raise RuntimeError(
270 'Cannot use call() when async_handlers is False.')
271 callback_event = self.eio.create_event()
272 callback_args = []
273
274 def event_callback(*args):
275 callback_args.append(args)
276 callback_event.set()
277
278 await self.emit(event, data=data, room=to or sid, namespace=namespace,
279 callback=event_callback, ignore_queue=ignore_queue)
280 try:
281 await asyncio.wait_for(callback_event.wait(), timeout)
282 except asyncio.TimeoutError:
283 raise exceptions.TimeoutError() from None
284 return callback_args[0] if len(callback_args[0]) > 1 \
285 else callback_args[0][0] if len(callback_args[0]) == 1 \
286 else None

632 async def _trigger_event(self, event, namespace, *args):
