31
32 :param logger: To enable logging set to ``True`` or pass a logger object to
33 use. To disable logging set to ``False``. The default is
34 ``False``. Note that fatal errors are logged even when
35 ``logger`` is ``False``.
36 :param json: An alternative json module to use for encoding and decoding
37 packets. Custom json modules must have ``dumps`` and ``loads``
38 functions that are compatible with the standard library
39 versions.
40 :param request_timeout: A timeout in seconds for requests. The default is
41 5 seconds.
42 :param http_session: an initialized ``requests.Session`` object to be used
43 when sending requests to the server. Use it if you
44 need to add special client options such as proxy
45 servers, SSL certificates, custom CA bundle, etc.
46 :param ssl_verify: ``True`` to verify SSL certificates, or ``False`` to
47 skip SSL certificate verification, allowing
48 connections to servers with self signed certificates.
49 The default is ``True``.
50 :param handle_sigint: Set to ``True`` to automatically handle disconnection
51 when the process is interrupted, or to ``False`` to
52 leave interrupt handling to the calling application.
53 Interrupt handling can only be enabled when the
54 client instance is created in the main thread.
55 :param websocket_extra_options: Dictionary containing additional keyword
56 arguments passed to

58 :param timestamp_requests: If ``True`` a timestamp is added to the query

234 def _connect_websocket(self, url, headers, engineio_path):
235 """Establish or upgrade to a WebSocket connection with the server."""
236 if websocket is None: # pragma: no cover
237 # not installed
238 self.logger.error('websocket-client package not installed, only '
239 'polling transport is available')
240 return False
241 websocket_url = self._get_engineio_url(url, engineio_path, 'websocket')
242 if self.sidsidsid:
243 self.logger.info(
244 'Attempting WebSocket upgrade to ' + websocket_url)
245 upgrade = True
246 websocket_url += '&sid=' + self.sidsidsid
247 else:
248 upgrade = False
249 self.base_urlbase_url = websocket_url
250 self.logger.info(
251 'Attempting WebSocket connection to ' + websocket_url)
252
253 # get cookies and other settings from the long-polling connection
254 # so that they are preserved when connecting to the WebSocket route
255 cookies = None
256 extra_options = {}
257 if self.httphttp:
258 # cookies
259 cookies = '; '.join([f"{cookie.name}={cookie.value}"
260 for cookie in self.httphttp.cookies])
261 for header, value in headers.items():
262 if header.lower() == 'cookie':
263 if cookies:
264 cookies += '; '
265 cookies += value
266 del headers[header]
267 break
268
269 # auth
270 if 'Authorization' not in headers and self.httphttp.auth is not None:
271 if not isinstance(self.httphttp.auth, tuple): # pragma: no cover
272 raise ValueError('Only basic authentication is supported')
273 basic_auth = '{}:{}'.format(
274 self.httphttp.auth[0], self.httphttp.auth[1]).encode('utf-8')
275 basic_auth = b64encode(basic_auth).decode('utf-8')
276 headers['Authorization'] = 'Basic ' + basic_auth
277
278 # cert
279 # this can be given as ('certfile', 'keyfile') or just 'certfile'
280 if isinstance(self.httphttp.cert, tuple):
281 extra_options['sslopt'] = {
282 'certfile': self.httphttp.cert[0],
283 'keyfile': self.httphttp.cert[1]}
284 elif self.httphttp.cert:
285 extra_options['sslopt'] = {'certfile': self.httphttp.cert}
286
287 # proxies
288 if self.httphttp.proxies:
289 proxy_url = None
290 if websocket_url.startswith('ws://'):
291 proxy_url = self.httphttp.proxies.get(
292 'ws', self.httphttp.proxies.get('http'))
293 else: # wss://
294 proxy_url = self.httphttp.proxies.get(
295 'wss', self.httphttp.proxies.get('https'))
296 if proxy_url:

298 proxy_url if '://' in proxy_url
299 else 'scheme://' + proxy_url)
300 extra_options['http_proxy_host'] = parsed_url.hostname
301 extra_options['http_proxy_port'] = parsed_url.port
302 extra_options['http_proxy_auth'] = (
303 (parsed_url.username, parsed_url.password)
304 if parsed_url.username or parsed_url.password
305 else None)
306
307 # verify
308 if isinstance(self.httphttp.verify, str):
309 if 'sslopt' in extra_options:
310 extra_options['sslopt']['ca_certs'] = self.httphttp.verify
311 else:
312 extra_options['sslopt'] = {'ca_certs': self.httphttp.verify}
313 elif not self.httphttp.verify:
314 self.ssl_verifyssl_verifyssl_verify = False
315
316 if not self.ssl_verifyssl_verifyssl_verify:
317 if 'sslopt' in extra_options:
318 extra_options['sslopt'].update({"cert_reqs": ssl.CERT_NONE})
319 else:
320 extra_options['sslopt'] = {"cert_reqs": ssl.CERT_NONE}
321
322 # combine internally generated options with the ones supplied by the
323 # caller. The caller's options take precedence.
324 headers.update(self.websocket_extra_optionswebsocket_extra_options.pop('header', {}))
325 extra_options['header'] = headers
326 extra_options['cookie'] = cookies
327 extra_options['enable_multithread'] = True
328 extra_options['timeout'] = self.request_timeout
329 extra_options.update(self.websocket_extra_optionswebsocket_extra_options)
330 try:
331 ws = websocket.create_connection(
332 websocket_url + self._get_url_timestamp(), **extra_options)
333 except (ConnectionError, OSError, websocket.WebSocketException):
334 if upgrade:
335 self.logger.warning(
336 'WebSocket upgrade failed: connection error')
337 return False
338 else:
339 raise exceptions.ConnectionError('Connection error')
340 if upgrade:
341 p = packet.Packet(packet.PING, data='probe').encode()

343 ws.send(p)
344 except Exception as e: # pragma: no cover
345 self.logger.warning(

347 str(e))
348 return False
349 try:
350 p = ws.recv()
351 except Exception as e: # pragma: no cover
352 self.logger.warning(
353 'WebSocket upgrade failed: unexpected recv exception: %s',
354 str(e))
355 return False
356 pkt = packet.Packet(encoded_packet=p)
357 if pkt.packet_type != packet.PONG or pkt.data != 'probe':
358 self.logger.warning(
359 'WebSocket upgrade failed: no PONG packet')
360 return False
361 p = packet.Packet(packet.UPGRADE).encode()
362 try:
363 ws.send(p)
364 except Exception as e: # pragma: no cover
365 self.logger.warning(
366 'WebSocket upgrade failed: unexpected send exception: %s',
367 str(e))
368 return False
369 self.current_transportcurrent_transport = 'websocket'
370 self.logger.info('WebSocket upgrade was successful')
371 else:
372 try:
373 p = ws.recv()
374 except Exception as e: # pragma: no cover
375 raise exceptions.ConnectionError(
376 'Unexpected recv exception: ' + str(e))

397 self._read_loop_websocket)

400 def _receive_packet(self, pkt):
401 """Handle incoming packets from the server."""
402 packet_name = packet.packet_names[pkt.packet_type] \
403 if pkt.packet_type < len(packet.packet_names) else 'UNKNOWN'
404 self.logger.info(
405 'Received packet %s data %s', packet_name,
406 pkt.data if not isinstance(pkt.data, bytes) else '<binary>')
407 if pkt.packet_type == packet.MESSAGE:
408 self._trigger_event('message', pkt.data, run_async=True)
409 elif pkt.packet_type == packet.PING:
410 self._send_packet(packet.Packet(packet.PONG, pkt.data))
411 elif pkt.packet_type == packet.CLOSE:
412 self.disconnect(abort=True, reason=self.reason.SERVER_DISCONNECT)
413 elif pkt.packet_type == packet.NOOP:
414 pass
415 else:
416 self.logger.error('Received unexpected packet of type %s',
417 pkt.packet_type)

557 def _write_loop(self):
558 """This background task sends packages to the server as they are
559 pushed to the send queue.
560 """
561 while self.statestate == 'connected':
562 # to simplify the timeout handling, use the maximum of the
563 # ping interval and ping timeout as timeout, with an extra 5
564 # seconds grace period
565 timeout = max(self.ping_intervalping_interval, self.ping_timeoutping_timeout) + 5
566 packets = None
567 try:
568 packets = [self.queuequeue.get(timeout=timeout)]
569 except self.queuequeue.Empty:
570 self.logger.error('packet queue is empty, aborting')
571 break
572 if packets == [None]:
573 self.queuequeue.task_done()
574 packets = []
575 else:
576 while True:
577 try:
578 packets.append(self.queuequeue.get(block=False))
579 except self.queuequeue.Empty:
580 break
581 if packets[-1] is None:
582 packets = packets[:-1]
583 self.queuequeue.task_done()
584 break
585 if not packets:
586 # empty packet list returned -> connection closed
587 break
588 if self.current_transportcurrent_transport == 'polling':
589 p = payload.Payload(packets=packets)
590 r = self._send_request(
591 'POST', self.base_urlbase_url, body=p.encode(),
592 headers={'Content-Type': 'text/plain'},
593 timeout=self.request_timeout)
594 for pkt in packets:
595 self.queuequeue.task_done()
596 if r is None or isinstance(r, str):
597 self.logger.warning(
598 r or 'Connection refused by the server, aborting')
599 break
600 if r.status_code < 200 or r.status_code >= 300:
601 self.logger.warning('Unexpected status code %s in server '
602 'response, aborting', r.status_code)
603 self.write_loop_taskwrite_loop_task = None
604 break
605 else:
606 # websocket
607 try:
608 for pkt in packets:
609 encoded_packet = pkt.encode()
610 if pkt.binary:
611 self.wsws.send_binary(encoded_packet)
612 else:
613 self.wsws.send(encoded_packet)
614 self.queuequeue.task_done()
615 except (websocket.WebSocketConnectionClosedException,
616 BrokenPipeError, OSError):
617 self.logger.warning(
618 'WebSocket connection was closed, aborting')
619 break
620 self.logger.info('Exiting write loop task')
