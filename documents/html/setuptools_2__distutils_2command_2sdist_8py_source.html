157 def run(self):
158 # 'filelist' contains the list of files that will make up the
159 # manifest
160 self.filelist = FileList()
161
162 # Run sub commands
163 for cmd_name in self.get_sub_commands():
164 self.run_command(cmd_name)
165
166 # Do whatever it takes to get the list of files to process
167 # (process the manifest template, read an existing manifest,
168 # whatever). File list is accumulated in 'self.filelist'.
169 self.get_file_list()
170
171 # If user just wanted us to regenerate the manifest, stop now.
172 if self.manifest_only:
173 return
174
175 # Otherwise, go ahead and create the source distribution tarball,
176 # or zipfile, or whatever.
177 self.make_distribution()

179 def get_file_list(self):
180 """Figure out the list of files to include in the source
181 distribution, and put it in 'self.filelist'. This might involve
182 reading the manifest template (and writing the manifest), or just
183 reading the manifest, or just using the default file set -- it all
184 depends on the user's options.
185 """
186 # new behavior when using a template:
187 # the file list is recalculated every time because
188 # even if MANIFEST.in or setup.py are not changed
189 # the user might have added some files in the tree that
190 # need to be included.
191 #
192 # This makes --force the default and only behavior with templates.
193 template_exists = os.path.isfile(self.template)
194 if not template_exists and self._manifest_is_not_generated():
195 self.read_manifest()
196 self.filelist.sort()
197 self.filelist.remove_duplicates()
198 return
199
200 if not template_exists:
201 self.warn(
202 ("manifest template '%s' does not exist " + "(using default file list)")
203 % self.template
204 )
205 self.filelist.findall()
206
207 if self.use_defaults:
208 self.add_defaults()
209
210 if template_exists:
211 self.read_template()
212
213 if self.prune:
214 self.prune_file_list()
215
216 self.filelist.sort()
217 self.filelist.remove_duplicates()
218 self.write_manifest()

224 - tests/test*.py and test/test*.py
225 - all pure Python modules mentioned in setup script

243 def _cs_path_exists(fspath):
244 """
245 Case-sensitive path existence check
246
247 >>> sdist._cs_path_exists(__file__)
248 True
249 >>> sdist._cs_path_exists(__file__.upper())
250 False
251 """
252 if not os.path.exists(fspath):
253 return False
254 # make absolute so we always have a directory
255 abspath = os.path.abspath(fspath)
256 directory, filename = os.path.split(abspath)
257 return filename in os.listdir(directory)

259 def _add_defaults_standards(self):
260 standards = [self.READMES, self.distribution.script_name]
261 for fn in standards:
262 if isinstance(fn, tuple):
263 alts = fn
264 got_it = False
265 for fn in alts:
266 if self._cs_path_exists(fn):
267 got_it = True
268 self.filelist.append(fn)
269 break
270
271 if not got_it:
272 self.warn(
273 "standard file not found: should have one of " + ', '.join(alts)
274 )
275 else:
276 if self._cs_path_exists(fn):
277 self.filelist.append(fn)
278 else:
279 self.warn(f"standard file '{fn}' not found")

335 def read_template(self):
336 """Read and parse manifest template file named by self.template.
337
338 (usually "MANIFEST.in") The parsing and processing is done by
339 'self.filelist', which updates itself accordingly.
340 """
341 log.info("reading manifest template '%s'", self.template)
342 template = TextFile(
343 self.template,
344 strip_comments=True,
345 skip_blanks=True,
346 join_lines=True,
347 lstrip_ws=True,
348 rstrip_ws=True,
349 collapse_join=True,
350 )
351
352 try:
353 while True:
354 line = template.readline()
355 if line is None: # end of file
356 break
357
358 try:
359 self.filelist.process_template_line(line)
360 # the call above can raise a DistutilsTemplateError for
361 # malformed lines, or a ValueError from the lower-level
362 # convert_path function
363 except (DistutilsTemplateError, ValueError) as msg:
364 self.warn(
365 "%s, line %d: %s"
366 % (template.filename, template.current_line, msg)
367 )
368 finally:
369 template.close()

371 def prune_file_list(self):
372 """Prune off branches that might slip into the file list as created
373 by 'read_template()', but really don't belong there:
374 * the build tree (typically "build")
375 * the release tree itself (only an issue if we ran "sdist"
376 previously with --keep-temp, or it aborted)
377 * any RCS, CVS, .svn, .hg, .git, .bzr, _darcs directories
378 """
379 build = self.get_finalized_command('build')
380 base_dir = self.distribution.get_fullname()
381
382 self.filelist.exclude_pattern(None, prefix=os.fspath(build.build_base))
383 self.filelist.exclude_pattern(None, prefix=base_dir)
384
385 if sys.platform == 'win32':
386 seps = r'/|\\'
387 else:
388 seps = '/'
389
390 vcs_dirs = ['RCS', 'CVS', r'\.svn', r'\.hg', r'\.git', r'\.bzr', '_darcs']
391 vcs_ptrn = r'(^|{})({})({}).*'.format(seps, '|'.join(vcs_dirs), seps)
392 self.filelist.exclude_pattern(vcs_ptrn, is_regex=True)

394 def write_manifest(self):
395 """Write the file list in 'self.filelist' (presumably as filled in
396 by 'add_defaults()' and 'read_template()') to the manifest file
397 named by 'self.manifest'.
398 """
399 if self._manifest_is_not_generated():
400 log.info(
401 f"not writing to manually maintained manifest file '{self.manifest}'"
402 )
403 return
404
405 content = self.filelist.files[:]
406 content.insert(0, '# file GENERATED by distutils, do NOT edit')
407 self.execute(
408 file_util.write_file,
409 (self.manifest, content),
410 f"writing manifest file '{self.manifest}'",
411 )

413 def _manifest_is_not_generated(self):
414 # check for special comment used in 3.1.3 and higher
415 if not os.path.isfile(self.manifest):
416 return False
417
418 with open(self.manifest, encoding='utf-8') as fp:
419 first_line = next(fp)
420 return first_line != '# file GENERATED by distutils, do NOT edit\n'

422 def read_manifest(self):
423 """Read the manifest file (named by 'self.manifest') and use it to
424 fill in 'self.filelist', the list of files to include in the source
425 distribution.
426 """
427 log.info("reading manifest file '%s'", self.manifest)
428 with open(self.manifest, encoding='utf-8') as lines:
429 self.filelist.extend(
430 # ignore comments and blank lines
431 filter(None, filterfalse(is_comment, map(str.strip, lines)))

434 def make_release_tree(self, base_dir, files):
435 """Create the directory tree that will become the source
436 distribution archive. All directories implied by the filenames in
437 'files' are created under 'base_dir', and then we hard link or copy
438 (if hard linking is unavailable) those files into place.
439 Essentially, this duplicates the developer's source tree, but in a
440 directory named after the distribution, containing only the files
441 to be distributed.
442 """
443 # Create all the directories under 'base_dir' necessary to
444 # put 'files' there; the 'mkpath()' is just so we don't die
445 # if the manifest happens to be empty.
446 self.mkpath(base_dir)
447 dir_util.create_tree(base_dir, files, dry_run=self.dry_run)
448
449 # And walk over the list of files, either making a hard link (if
450 # os.link exists) to each one that doesn't already exist in its
451 # corresponding location under 'base_dir', or copying each file
452 # that's out-of-date in 'base_dir'. (Usually, all files will be
453 # out-of-date, because by default we blow away 'base_dir' when
454 # we're done making the distribution archives.)
455
456 if hasattr(os, 'link'): # can make hard links on this system
457 link = 'hard'
458 msg = f"making hard links in {base_dir}..."
459 else: # nope, have to copy
460 link = None
461 msg = f"copying files to {base_dir}..."
462
463 if not files:
464 log.warning("no files to distribute -- empty manifest?")
465 else:
466 log.info(msg)
467 for file in files:
468 if not os.path.isfile(file):
469 log.warning("'%s' not a regular file -- skipping", file)
470 else:
471 dest = os.path.join(base_dir, file)
472 self.copy_file(file, dest, link=link)
473
474 self.distribution.metadata.write_pkg_info(base_dir)

476 def make_distribution(self):
477 """Create the source distribution(s). First, we create the release
478 tree with 'make_release_tree()'; then, we create all required
479 archive files (according to 'self.formats') from the release tree.
480 Finally, we clean up by blowing away the release tree (unless
481 'self.keep_temp' is true). The list of archive files created is
482 stored so it can be retrieved later by 'get_archive_files()'.
483 """
484 # Don't warn about missing meta-data here -- should be (and is!)
485 # done elsewhere.
486 base_dir = self.distribution.get_fullname()
487 base_name = os.path.join(self.dist_dir, base_dir)
488
489 self.make_release_tree(base_dir, self.filelist.files)
490 archive_files = [] # remember names of files we create
491 # tar archive must be created last to avoid overwrite and remove
492 if 'tar' in self.formats:
493 self.formats.append(self.formats.pop(self.formats.index('tar')))
494
495 for fmt in self.formats:
496 file = self.make_archive(
497 base_name, fmt, base_dir=base_dir, owner=self.owner, group=self.group
498 )
499 archive_files.append(file)
500 self.distribution.dist_files.append(('sdist', '', file))
501
502 self.archive_files = archive_files
503
504 if not self.keep_temp:
505 dir_util.remove_tree(base_dir, dry_run=self.dry_run)
